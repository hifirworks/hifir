144  template <typename T = hif_lapack_int>
145  inline static typename std::enable_if<!_INT_TYPE_CONSIS, T>::type geqp3(
146  const int_type m, const int_type n, pointer a, const int_type lda,
147  int_type *jpvt, pointer tau, pointer work, const int_type lwork,
148  scalar_type *rwork) {
149  if (lwork == (int_type)-1)
150  return internal::geqp3(m, n, a, lda, nullptr, tau, work, -1, rwork);
151  // NOTE jpvt can be input as well
152  std::vector<hif_lapack_int> buf(jpvt, jpvt + n);
153  const auto info =
154  internal::geqp3(m, n, a, lda, buf.data(), tau, work, lwork, rwork);
155  std::copy(buf.cbegin(), buf.cend(), jpvt);
156  return info;
157  }

159  inline static int_type geqp3(DenseMatrix<value_type> &a,
160  Array<int_type> &jpvt, Array<value_type> &tau) {
161  // query opt size
162  hif_assert(jpvt.size() == a.ncols(),
163  "column pivoting should have size of column");
164  hif_assert(tau.size() == std::min(a.nrows(), a.ncols()),
165  "tau should have size of min(m,n)");
166  value_type lwork;
167  geqp3(a.nrows(), a.ncols(), a.data(), a.nrows(), jpvt.data(), tau.data(),
168  &lwork, -1, nullptr);
169  std::vector<value_type> work((int_type)abs(lwork));
170  Array<scalar_type> rwork;
171  if (!IS_REAL) rwork.resize(2 * a.ncols());
172  return geqp3(a.nrows(), a.ncols(), a.data(), a.nrows(), jpvt.data(),
173  tau.data(), work.data(), (int_type)abs(lwork), rwork.data());
174  }

194  inline static int_type trcon(const char norm, const char uplo,
195  const char diag,
196  const DenseMatrix<value_type> &a,
197  value_type & rcond) {
198  hif_assert(a.nrows() >= a.ncols(),
199  "a must have row size that is no smaller than its column size");
200  std::vector<value_type> work(3 * a.ncols());
201  std::vector<hif_lapack_int> iwork(a.ncols());
202  return trcon(norm, uplo, diag, a.ncols(), a.data(), a.nrows(), rcond,
203  work.data(), iwork.data());
